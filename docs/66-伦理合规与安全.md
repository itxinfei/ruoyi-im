```python
# ai_risk_classifier.py
from enum import Enum

class AIRiskLevel(Enum):
    """AI系统风险等级 / AI System Risk Level"""
    UNACCEPTABLE = "unacceptable"   # 不可接受风险（禁止）
    HIGH = "high"                   # 高风险（严格监管）
    LIMITED = "limited"             # 有限风险（透明度要求）
    MINIMAL = "minimal"             # 最小风险（自愿遵守）

RISK_MAPPING = {
    # 不可接受风险 - 禁止使用
    "social_scoring": AIRiskLevel.UNACCEPTABLE,
    "real_time_biometric": AIRiskLevel.UNACCEPTABLE,
    "manipulation": AIRiskLevel.UNACCEPTABLE,

    # 高风险 - 需要严格评估
    "critical_infrastructure": AIRiskLevel.HIGH,
    "education_assessment": AIRiskLevel.HIGH,
    "employment_hiring": AIRiskLevel.HIGH,
    "credit_scoring": AIRiskLevel.HIGH,
    "medical_devices": AIRiskLevel.HIGH,

    # 有限风险 - 透明度要求
    "chatbot": AIRiskLevel.LIMITED,
    "generative_ai": AIRiskLevel.LIMITED,
    "deepfake": AIRiskLevel.LIMITED,

    # 最小风险
    "spam_filter": AIRiskLevel.MINIMAL,
    "recommendation": AIRiskLevel.MINIMAL,
    "game_ai": AIRiskLevel.MINIMAL
}

def classify_risk(use_case: str) -> AIRiskLevel:
    """
    分类AI系统风险等级

    参数说明 (Parameters):
        use_case: str - 使用场景 (Use case description)

    返回值 (Return Value):
        AIRiskLevel - 风险等级
    """
    use_case_lower = use_case.lower()

    for key, level in RISK_MAPPING.items():
        if key in use_case_lower:
            return level

    return AIRiskLevel.LIMITED  # 默认有限风险
```

### 2.2 隐私影响评估 (PIA)

```python
# privacy_impact_assessment.py
from typing import List, Dict, Optional
from dataclasses import dataclass
from enum import Enum

class PrivacyRisk(Enum):
    """隐私风险等级 / Privacy Risk Level"""
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"
    CRITICAL = "critical"

@dataclass
class DataProcessingActivity:
    """数据处理活动 / Data Processing Activity"""
    activity_id: str
    activity_name: str
    data_types: List[str]
    purposes: List[str]
    legal_basis: str
    recipients: List[str]
    retention_period: str
    safeguards: List[str]

class PrivacyImpactAssessment:
    def __init__(self):
        self.risk_factors = {
            "data_volume": {"weight": 0.2, "thresholds": [1000, 10000, 100000]},
            "sensitivity": {"weight": 0.3, "thresholds": [1, 2, 3]},
            "automated_decision": {"weight": 0.25, "thresholds": [0, 1, 2]},
            "cross_border": {"weight": 0.15, "thresholds": [0, 1, 2]},
            "new_technology": {"weight": 0.1, "thresholds": [0, 1, 2]}
        }

    def assess_risk(self, activity: DataProcessingActivity) -> Dict:
        """
        评估隐私风险

        参数说明 (Parameters):
            activity: DataProcessingActivity - 数据处理活动

        返回值 (Return Value):
            dict - 风险评估结果
        """
        scores = {}

        # 数据量评分
        data_volume_score = self._score_by_thresholds(
            len(activity.data_types), self.risk_factors["data_volume"]["thresholds"]
        )
        scores["data_volume"] = data_volume_score

        # 敏感度评分
        sensitivity_mapping = {"public": 0, "internal": 1, "confidential": 2, "top_secret": 3}
        sensitivity_score = max(sensitivity_mapping.get(dt, 1) for dt in activity.data_types)
        scores["sensitivity"] = sensitivity_score

        # 综合评分
        total_score = sum(
            scores[k] * self.risk_factors[k]["weight"]
            for k in scores
        )

        # 风险等级判定
        if total_score >= 0.7:
            risk_level = PrivacyRisk.CRITICAL
        elif total_score >= 0.5:
            risk_level = PrivacyRisk.HIGH
        elif total_score >= 0.3:
            risk_level = PrivacyRisk.MEDIUM
        else:
            risk_level = PrivacyRisk.LOW

        return {
            "risk_level": risk_level,
            "total_score": round(total_score, 2),
            "component_scores": scores,
            "recommendations": self._get_recommendations(risk_level)
        }

    def _score_by_thresholds(self, value: int, thresholds: List[int]) -> int:
        """根据阈值评分"""
        for i, threshold in enumerate(thresholds):
            if value < threshold:
                return i
        return len(thresholds)

    def _get_recommendations(self, risk_level: PrivacyRisk) -> List[str]:
        """获取风险缓解建议"""
        recommendations = {
            PrivacyRisk.LOW: ["记录处理活动", "定期审查"],
            PrivacyRisk.MEDIUM: ["实施访问控制", "加密存储", "定期安全评估"],
            PrivacyRisk.HIGH: ["进行正式PIA", "引入DPO审查", "建立数据保护机制"],
            PrivacyRisk.CRITICAL: ["暂停处理", "全面影响评估", "监管机构咨询"]
        }
        return recommendations[risk_level]
```

### 2.3 数据主体权利保障

```python
# data_subject_rights.py
from typing import List, Dict, Optional
from dataclasses import dataclass
from datetime import datetime, timedelta

@dataclass
class DataSubjectRequest:
    """数据主体请求 / Data Subject Request"""
    request_id: str
    requester_id: str
    request_type: str           # access, rectification, erasure, portability, objection
    submitted_at: datetime
    status: str                 # pending, processing, completed, denied
    deadline: datetime          # 30天内响应

class DataSubjectRightsManager:
    def __init__(self):
        self.requests: List[DataSubjectRequest] = []
        self.response_templates = self._init_templates()

    def _init_templates(self) -> Dict:
        """初始化响应模板"""
        return {
            "access": {
                "title": "数据访问响应 / Data Access Response",
                "description": "提供请求者个人数据的完整副本"
            },
            "rectification": {
                "title": "数据更正响应 / Data Rectification Response",
                "description": "更正不准确的个人数据"
            },
            "erasure": {
                "title": "数据删除响应 / Data Erasure Response",
                "description": "删除请求者的个人数据（保留例外情况）"
            },
            "portability": {
                "title": "数据可携响应 / Data Portability Response",
                "description": "提供结构化、机器可读的格式数据"
            },
            "objection": {
                "title": "反对处理响应 / Objection Processing Response",
                "description": "处理请求者对特定数据处理的反对"
            }
        }

    def submit_request(
        self,
        requester_id: str,
        request_type: str
    ) -> DataSubjectRequest:
        """
        提交数据主体请求

        参数说明 (Parameters):
            requester_id: str - 请求者ID (Requester ID)
            request_type: str - 请求类型 (Request type)

        返回值 (Return Value):
            DataSubjectRequest - 创建的请求对象
        """
        request = DataSubjectRequest(
            request_id=f"DSR-{len(self.requests) + 1:06d}",
            requester_id=requester_id,
            request_type=request_type,
            submitted_at=datetime.now(),
            status="pending",
            deadline=datetime.now() + timedelta(days=30)
        )
        self.requests.append(request)
        return request

    def process_request(
        self,
        request_id: str,
        action: str,
        data: Optional[Dict] = None
    ) -> Dict:
        """
        处理数据主体请求

        参数说明 (Parameters):
            request_id: str - 请求ID (Request ID)
            action: str - 处理动作 (Processing action: approve, deny, partial)
            data: Optional[Dict] - 相关数据 (Related data)

        返回值 (Return Value):
            dict - 处理结果
        """
        request = next((r for r in self.requests if r.request_id == request_id), None)

        if not request:
            return {"success": False, "error": "Request not found"}

        request.status = "completed" if action == "approve" else "denied"

        return {
            "success": True,
            "request_id": request_id,
            "action": action,
            "completed_at": datetime.now().isoformat(),
            "response_template": self.response_templates.get(request.request_type)
        }
```

## 三、内容安全 (Content Safety)

### 3.1 安全分类体系

```python
# content_classifier.py
from enum import Enum
from typing import Dict, List

class ContentCategory(Enum):
    """内容安全分类 / Content Safety Categories"""
    SAFE = "safe"
    SEXUAL = "sexual"
    VIOLENCE = "violence"
    HATE = "hate"
    SELF_HARM = "self_harm"
    TERRORISM = "terrorism"
    ILLEGAL = "illegal"
    HARASSMENT = "harassment"
    MISINFORMATION = "misinformation"

class ContentSafetyLevel(Enum):
    """内容安全等级 / Content Safety Levels"""
    SAFE = 0           # 安全
    CAUTION = 1        # 需注意
    WARNING = 2        # 警告
    BLOCK = 3          # 阻止

SAFETY_THRESHOLDS = {
    ContentCategory.SEXUAL: 0.7,
    ContentCategory.VIOLENCE: 0.7,
    ContentCategory.HATE: 0.6,
    ContentCategory.SELF_HARM: 0.5,
    ContentCategory.TERRORISM: 0.5,
    ContentCategory.ILLEGAL: 0.6,
    ContentCategory.HARASSMENT: 0.6,
    ContentCategory.MISINFORMATION: 0.5
}

class ContentClassifier:
    def __init__(self, model_path: str = None):
        """
        初始化内容分类器

        参数说明 (Parameters):
            model_path: Optional[str] - 分类模型路径 (Classifier model path)
        """
        # 实际应用中加载训练好的分类模型
        self.model = None

    def classify(
        self,
        text: str,
        threshold: float = 0.5
    ) -> Dict:
        """
        内容安全分类

        参数说明 (Parameters):
            text: str - 待分类文本 (Text to classify)
            threshold: float - 分类阈值 (Classification threshold) Default: 0.5

        返回值 (Return Value):
            dict - 分类结果
        """
        # 实际应用中调用模型推理
        # 此处为简化示例
        scores = self._predict_scores(text)

        # 判断安全等级
        max_score = max(scores.values()) if scores else 0
        max_category = max(scores, key=scores.get) if scores else ContentCategory.SAFE

        # 应用阈值
        for category, category_threshold in SAFETY_THRESHOLDS.items():
            if scores.get(category.value, 0) >= category_threshold:
                return {
                    "is_safe": False,
                    "category": category.value,
                    "score": scores.get(category.value, 0),
                    "level": ContentSafetyLevel.BLOCK.value,
                    "action": "block"
                }

        if max_score >= threshold:
            return {
                "is_safe": False,
                "category": max_category.value if isinstance(max_category, ContentCategory) else max_category,
                "score": max_score,
                "level": ContentSafetyLevel.CAUTION.value,
                "action": "caution"
            }

        return {
            "is_safe": True,
            "category": "safe",
            "score": 1.0 - max_score,
            "level": ContentSafetyLevel.SAFE.value,
            "action": "pass"
        }

    def _predict_scores(self, text: str) -> Dict:
        """预测各分类分数（简化示例）"""
        # 实际应用中应使用真实模型
        return {
            "safe": 0.95,
            "sexual": 0.05,
            "violence": 0.02
        }
```

## 四、模型安全 (Model Security)

### 4.1 对抗攻击防护

```python
# adversarial_defense.py
from typing import Dict, List, Optional
import numpy as np

class AdversarialDefense:
    def __init__(self):
        self.defense_config = {
            "input_validation": True,
            "prompt_injection_detection": True,
            "jailbreak_detection": True,
            "toxicity_filter": True
        }

    def detect_prompt_injection(self, prompt: str) -> Dict:
        """
        检测提示词注入攻击

        参数说明 (Parameters):
            prompt: str - 用户输入提示词 (User input prompt)

        返回值 (Return Value):
            dict - 检测结果
        """
        injection_patterns = [
            r"ignore\s+previous\s+instructions",
            r"disregard\s+.*\s+rules",
            r"you\s+are\s+now\s+",
            r"forget\s+.*\s+everything",
            r"new\s+system\s+prompt",
            r"override\s+",
            r"\\x00|\\n|\\t"  # 转义字符注入
        ]

        matches = []
        for pattern in injection_patterns:
            import re
            if re.search(pattern, prompt, re.IGNORECASE):
                matches.append(pattern)

        risk_score = len(matches) / len(injection_patterns)

        return {
            "is_injection": risk_score > 0.3,
            "risk_score": risk_score,
            "matched_patterns": matches,
            "recommendation": "block" if risk_score > 0.5 else "sanitize"
        }

    def detect_jailbreak(self, prompt: str, context: Optional[Dict] = None) -> Dict:
        """
        检测越狱攻击

        参数说明 (Parameters):
            prompt: str - 用户输入
            context: Optional[Dict] - 上下文信息

        返回值 (Return Value):
            dict - 检测结果
        """
        jailbreak_indicators = {
            "role_play": ["pretend", "act as", "role of", "imagine"],
            "hypothetical": ["what if", "hypothetically", "in a fictional"],
            "safety_bypass": [" DAN", "do anything now", "developer mode"],
            "encoding": ["base64", "hex", "unicode escape"]
        }

        detected_categories = []
        for category, keywords in jailbreak_indicators.items():
            if any(kw in prompt.lower() for kw in keywords):
                detected_categories.append(category)

        risk_level = "high" if len(detected_categories) >= 2 else "medium" if detected_categories else "low"

        return {
            "is_jailbreak": risk_level == "high",
            "risk_level": risk_level,
            "detected_categories": detected_categories,
            "action": "block" if risk_level == "high" else "monitor"
        }

    def sanitize_input(self, prompt: str) -> str:
        """
        输入净化处理

        参数说明 (Parameters):
            prompt: str - 原始输入

        返回值 (Return Value):
            str - 净化后的输入
        """
        import re

        # 移除潜在危险字符
        sanitized = prompt
        sanitized = re.sub(r'[\x00-\x08\x0b\x0c\x0e-\x1f]', '', sanitized)
        sanitized = re.sub(r'[\u200b-\u200f\u2028-\u202f]', '', sanitized)

        return sanitized
```

### 4.2 模型水印

```python
# model_watermark.py
import hashlib
import numpy as np

class ModelWatermark:
    def __init__(self):
        self.watermark_pattern = self._generate_pattern()

    def _generate_pattern(self) -> np.ndarray:
        """生成水印模式"""
        np.random.seed(42)
        return np.random.randn(10)

    def embed_watermark(self, text: str) -> str:
        """
        在生成的文本中嵌入水印

        参数说明 (Parameters):
            text: str - 原始生成文本

        返回值 (Return Value):
            str - 嵌入水印的文本
        """
        # 简化的水印嵌入方法
        # 实际应用中应使用更复杂的统计水印技术
        watermark_hash = hashlib.sha256(
            (text + "watermark_key").encode()
        ).hexdigest()[:8]

        return f"{text}\n<!--watermark:{watermark_hash}-->"

    def detect_watermark(self, text: str) -> bool:
        """
        检测文本中的水印

        参数说明 (Parameters):
            text: str - 待检测文本

        返回值 (Return Value):
            bool - 是否包含水印
        """
        return "<!--watermark:" in text
```
