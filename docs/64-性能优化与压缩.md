## 概述 (Overview)

本文档涵盖模型性能优化的核心技木，包括模型量化、剪枝、知识蒸馏等压缩技术，以及推理加速优化手段，帮助在保证效果的前提下提升推理效率。

## 一、模型量化 (Model Quantization)

### 1.1 量化技术对比

| 量化方法 | 精度 | 显存节省 | 速度提升 | 效果损失 | 适用场景 |
|---------|------|---------|---------|---------|---------|
| FP16 | 16位 | 50% | 1.2x | 极小 | 通用场景 |
| BF16 | 16位 | 50% | 1.1x | 极小 | 训练/推理 |
| INT8 | 8位 | 75% | 2-4x | 小 | 推理加速 |
| INT4 | 4位 | 87.5% | 4-8x | 中等 | 极致压缩 |
| NF4 | 4位 | 87.5% | 4-8x | 较小 | QLoRA |

### 1.2 动态量化 (Post-Training Dynamic Quantization)

```python
# dynamic_quantization.py
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

def dynamic_quantize_int8(
    model_path: str,
    output_path: str
):
    """
    动态量化（无需重训练）

    参数说明 (Parameters):
        model_path: str - 输入模型路径 (Input model path)
        output_path: str - 输出模型路径 (Output model path)

    适用场景 (Use Cases):
        - 快速部署
        - 无训练数据
        - 效果优先
    """
    # 加载模型
    model = AutoModelForCausalLM.from_pretrained(
        model_path,
        torch_dtype=torch.float32,
        trust_remote_code=True
    )

    # 动态量化（权重int8，激活fp32）
    quantized_model = torch.quantization.quantize_dynamic(
        model,
        {torch.nn.Linear},  # 量化的层类型
        dtype=torch.qint8
    )

    # 保存
    quantized_model.save_pretrained(output_path)

    # 对比大小
    original_size = sum(p.numel() * p.element_size() for p in model.parameters())
    quantized_size = sum(p.numel() * p.element_size() for p in quantized_model.parameters())

    print(f"原始大小: {original_size / 1024**2:.2f} MB")
    print(f"量化后: {quantized_size / 1024**2:.2f} MB")
    print(f"压缩比: {original_size / quantized_size:.2f}x")

    return {
        "original_mb": original_size / 1024**2,
        "quantized_mb": quantized_size / 1024**2,
        "compression_ratio": original_size / quantized_size
    }
```

### 1.3 静态量化 (Post-Training Static Quantization)

```python
# static_quantization.py
import torch
from torch.quantization importquantize_fx
from transformers import AutoModelForCausalLM

def static_quantize_int8(
    model_path: str,
    calibration_data: list,
    output_path: str
):
    """
    静态量化（需要校准数据）

    参数说明 (Parameters):
        model_path: str - 输入模型路径 (Input model path)
        calibration_data: list - 校准数据集 (Calibration dataset)
        output_path: str - 输出模型路径 (Output model path)

    步骤 (Steps):
        1. 准备校准数据
        2. 执行模型量化
        3. 保存量化模型
    """
    # 加载模型
    model = AutoModelForCausalLM.from_pretrained(
        model_path,
        torch_dtype=torch.float32,
        trust_remote_code=True
    )
    model.eval()

    # 准备校准函数
    def calibrate(model, calib_loader):
        """校准：收集激活值统计"""
        with torch.no_grad():
            for batch in calib_loader:
                model(batch)

    # 量化配置
    qconfig = torch.quantization.get_default_qconfig('fbgemm')
    torch.backends.quantized.engine = 'fbgemm'

    # 准备量化模型
    model_prepared = torch.quantization.prepare(
        model,
        inplace=False
    )

    # 校准（通常100-1000个样本即可）
    calibrate(model_prepared, calibration_data)

    # 执行量化
    quantized_model = torch.quantization.convert(
        model_prepared,
        inplace=False
    )

    # 保存
    quantized_model.save_pretrained(output_path)
```

```python
# gptq_quantization.py
from auto_gptq import AutoGPTQForCausalLM, BaseQuantizeConfig
from transformers import AutoTokenizer

def gptq_quantize(
    model_path: str,
    output_path: str,
    quantization_bit: int = 4,
    calibration_data: list = None
):
    """
    GPTQ量化

    参数说明 (Parameters):
        model_path: str - 输入模型路径 (Input model path)
        output_path: str - 输出模型路径 (Output model path)
        quantization_bit: int - 量化位数 (Quantization bits) Default: 4
        calibration_data: list - 校准数据 (Calibration data)

    优势 (Advantages):
        - 4bit量化效果接近fp16
        - 推理速度接近fp16
        - 显存占用大幅降低
    """
    tokenizer = AutoTokenizer.from_pretrained(model_path)
    quantize_config = BaseQuantizeConfig(
        bits=quantization_bit,
        desc_act=True,  # 激活顺序量化，效果更好
        group_size=128  # 每组参数量
    )

    model = AutoGPTQForCausalLM.from_pretrained(
        model_path,
        quantize_config=quantize_config
    )

    # 校准（如果提供校准数据）
    if calibration_data:
        model.quantize(calibration_data)

    # 保存
    model.save_quantized(output_path)
    tokenizer.save_pretrained(output_path)
```

## 二、模型剪枝 (Model Pruning)

### 2.1 剪枝策略

| 剪枝类型 | 说明 | 效果 | 适用场景 |
|---------|-----|------|---------|
| 非结构化剪枝 | 随机删除任意权重 | 中等压缩 |理论研究 |
| 结构化剪枝 | 按行/列/头删除 | 高效硬件友好 | 工程部署 |
| 动态剪枝 | 根据输入动态稀疏 | 最优压缩率 | 推理加速 |
| 渐进式剪枝 | 逐步增加剪枝率 | 稳定收敛 | 效果保障 |

### 2.2 结构化剪枝

```python
# structured_pruning.py
import torch
from transformers import AutoModelForCausalLM
import torch.nn.utils.prune as prune

def structured_pruning(
    model_path: str,
    output_path: str,
    sparsity: float = 0.3,
    prune_method: str = "l1"
):
    """
    结构化剪枝

    参数说明 (Parameters):
        model_path: str - 输入模型路径 (Input model path)
        output_path: str - 输出模型路径 (Output model path)
        sparsity: float - 剪枝比例 (Pruning ratio) Default: 0.3 (30%)
        prune_method: str - 剪枝方法(l1/ln/random) (Pruning method) Default: "l1"

    支持的剪枝方法 (Supported Methods):
        - l1: L1范数重要性
        - ln: Ln范数重要性
        - random: 随机剪枝
    """
    model = AutoModelForCausalLM.from_pretrained(
        model_path,
        trust_remote_code=True
    )

    # 获取可剪枝的参数
    parameters_to_prune = []
    for name, module in model.named_modules():
        if isinstance(module, torch.nn.Linear):
            parameters_to_prune.append((module, 'weight'))

    # 执行结构化剪枝（按模块级别）
    for module, name in parameters_to_prune:
        if prune_method == "l1":
            prune.l1_unstructured(
                module,
                name=name,
                amount=sparsity
            )
        elif prune_method == "ln":
            prune.ln_structured(
                module,
                name=name,
                amount=sparsity,
                n=2,
                dim=0
            )

    # 移除剪枝掩码（永久化）
    for module, name in parameters_to_prune:
        prune.remove(module, name)

    # 保存
    model.save_pretrained(output_path)

    # 计算稀疏度
    total_params = 0
    zero_params = 0
    for p in model.parameters():
        total_params += p.numel()
        zero_params += (p == 0).sum().item()

    sparsity_ratio = zero_params / total_params
    print(f"实际稀疏度: {sparsity_ratio:.2%}")
```

### 2.3 Attention Head剪枝

```python
# attention_head_pruning.py
import torch
from transformers import AutoModelForCausalLM
import numpy as np

def prune_attention_heads(
    model,
    importance_scores: dict,
    num_heads_to_prune: int
):
    """
    剪枝Attention Head

    参数说明 (Parameters):
        model: nn.Module - 模型实例
        importance_scores: dict - 各head的重要性分数
        num_heads_to_prune: int - 需要剪枝的头数量

    重要性计算方法 (Importance Calculation):
        - 基于注意力熵
        - 基于梯度
        - 基于注意力流
    """
    heads_to_prune = {}

    for layer_name, scores in importance_scores.items():
        # 获取最不重要的head索引
        sorted_indices = np.argsort(scores)
        prune_indices = sorted_indices[:num_heads_to_prune].tolist()
        heads_to_prune[layer_name] = prune_indices

    # 应用剪枝
    model.prune_heads(heads_to_prune)

    return model
```

## 三、知识蒸馏 (Knowledge Distillation)

### 3.1 蒸馏策略

| 蒸馏类型 | 教师模型 | 学生模型 | 效果 | 复杂度 |
|---------|---------|---------|------|--------|
| Response Distillation | 7B | 1B | 中等 | 低 |
| Feature Distillation | 7B | 1B | 良好 | 中 |
| Chain-of-Thought | 7B+CoT | 1B | 良好 | 高 |
| Multi-Token Prediction | 7B | 1B | 优秀 | 高 |

### 3.2 Response Distillation

```python
# knowledge_distillation.py
import torch
import torch.nn as nn
from torch.utils.data import DataLoader
from transformers import AutoModelForCausalLM, AutoTokenizer

class DistillationLoss(nn.Module):
    def __init__(self, temperature: float = 2.0, alpha: float = 0.5):
        super().__init__()
        self.temperature = temperature
        self.alpha = alpha
        self.ce_loss = nn.CrossEntropyLoss()
        self.kl_loss = nn.KLDivLoss(reduction="batchmean")

    def forward(
        self,
        student_logits: torch.Tensor,
        teacher_logits: torch.Tensor,
        labels: torch.Tensor
    ) -> torch.Tensor:
        """
        蒸馏损失计算

        参数说明 (Parameters):
            student_logits: Tensor - 学生模型logits
            teacher_logits: Tensor - 教师模型logits
            labels: Tensor - 真实标签

        损失组成 (Loss Components):
            - ce_loss: 交叉熵损失
            - kl_loss: 知识蒸馏KL散度损失
        """
        # 软标签损失
        soft_teacher = torch.softmax(teacher_logits / self.temperature, dim=-1)
        soft_student = torch.log_softmax(student_logits / self.temperature, dim=-1)
        kl_loss = self.kl_loss(soft_student, soft_teacher) * (self.temperature ** 2)

        # 硬标签损失
        ce_loss = self.ce_loss(
            student_logits.view(-1, student_logits.size(-1)),
            labels.view(-1)
        )

        # 加权组合
        total_loss = self.alpha * kl_loss + (1 - self.alpha) * ce_loss
        return total_loss

def distill(
    teacher_model_path: str,
    student_model_path: str,
    train_data: DataLoader,
    output_path: str,
    temperature: float = 2.0,
    alpha: float = 0.5,
    epochs: int = 3
):
    """
    知识蒸馏训练

    参数说明 (Parameters):
        teacher_model_path: str - 教师模型路径 (Teacher model path)
        student_model_path: str - 学生模型路径 (Student model path)
        train_data: DataLoader - 训练数据加载器
        output_path: str - 输出路径 (Output path)
        temperature: float - 蒸馏温度 (Distillation temperature) Default: 2.0
        alpha: float - 损失权重 (Loss weight) Default: 0.5
        epochs: int - 训练轮数 (Training epochs) Default: 3

    返回值 (Return Value):
        nn.Module - 训练好的学生模型
    """
    # 加载教师模型（冻结）
    teacher_model = AutoModelForCausalLM.from_pretrained(
        teacher_model_path,
        trust_remote_code=True
    ).eval()

    # 加载学生模型
    student_model = AutoModelForCausalLM.from_pretrained(
        student_model_path,
        trust_remote_code=True
    )

    # 损失函数
    criterion = DistillationLoss(temperature=temperature, alpha=alpha)
    optimizer = torch.optim.AdamW(student_model.parameters(), lr=1e-5)

    # 训练循环
    for epoch in range(epochs):
        for batch in train_data:
            # 教师预测（不计算梯度）
            with torch.no_grad():
                teacher_outputs = teacher_model(**batch)

            # 学生预测
            student_outputs = student_model(**batch)

            # 计算蒸馏损失
            loss = criterion(
                student_outputs.logits,
                teacher_outputs.logits,
                batch["labels"]
            )

            # 反向传播
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

    # 保存学生模型
    student_model.save_pretrained(output_path)
    return student_model
```

### 4.1 Flash Attention

```python
# flash_attention.py
from transformers import AutoModelForCausalLM, AutoConfig

def enable_flash_attention(
    model_path: str,
    output_path: str = None
):
    """
    启用Flash Attention加速

    参数说明 (Parameters):
        model_path: str - 模型路径 (Model path)
        output_path: str - 输出路径 (Optional, defaults to same as input)

    加速效果 (Speedup):
        - 显存: 减少30-50%
        - 速度: 提升2-4x

    注意事项 (Notes):
        - 需要A100/H100等支持Flash Attention的GPU
        - CUDA >= 11.8
    """
    config = AutoConfig.from_pretrained(model_path, trust_remote_code=True)

    # 启用Flash Attention
    config.attn_config = {
        "attn_type": "flash",
        "attn_probability": None
    }

    model = AutoModelForCausalLM.from_pretrained(
        model_path,
        config=config,
        trust_remote_code=True
    )

    if output_path:
        model.save_pretrained(output_path)

    return model
```

```python
import torch.distributed as dist

def init_tensor_parallel(tp_size: int = 2):
    """
    初始化Tensor并行

    参数说明 (Parameters):
        tp_size: int - 并行度 (Parallelism degree) Default: 2

        - 单卡显存不足
        - 需要更大batch_size
        - 多卡推理加速
    """

    dist.init_process_group(backend="nccl")

    # 设置当前设备
    local_rank = int(os.environ["LOCAL_RANK"])

    return local_rank

def load_model_tensor_parallel(
    model_path: str,
    tp_size: int
):
    """
    加载支持Tensor并行的模型

    参数说明 (Parameters):
        model_path: str - 模型路径
        tp_size: int - 并行度

    显存计算 (Memory Calculation):
        单卡显存 = 模型大小 / tp_size + KV缓存
    """

        checkpoint=model_path,
        device_map="auto",
        no_split_module_classes=["LlamaDecoderLayer"]
    )

    return model
```

### 4.3 批处理优化

```python
# batch_optimization.py
from vllm import LLM, SamplingParams
from typing import List

class DynamicBatching:
    def __init__(self, llm: LLM):
        self.llm = llm
        self.pending_requests: List[dict] = []
        self.max_wait_time_ms: int = 100

    def add_request(
        self,
        prompt: str,
        max_tokens: int = 256,
        temperature: float = 0.7
    ) -> str:
        """
        添加推理请求

        参数说明 (Parameters):
            prompt: str - 输入提示词 (Input prompt)
            max_tokens: int - 最大生成长度 (Max tokens to generate) Default: 256
            temperature: float - 采样温度 (Sampling temperature) Default: 0.7

        返回值 (Return Value):
            str - 请求ID，用于获取结果
        """
        request_id = f"req_{len(self.pending_requests)}"
        self.pending_requests.append({
            "id": request_id,
            "prompt": prompt,
            "params": SamplingParams(
                max_tokens=max_tokens,
                temperature=temperature
            )
        })
        return request_id

    def execute_batch(self) -> List[dict]:
        """
        执行批量推理

        返回值 (Return Value):
            List[dict] - 推理结果列表
        """
        if not self.pending_requests:
            return []

        prompts = [req["prompt"] for req in self.pending_requests]
        params_list = [req["params"] for req in self.pending_requests]

        # vLLM自动动态批处理
        outputs = self.llm.generate(prompts)

        results = []
        for i, output in enumerate(outputs):
            results.append({
                "id": self.pending_requests[i]["id"],
                "text": output.outputs[0].text,
                "stop_reason": output.outputs[0].finish_reason
            })

        self.pending_requests.clear()
        return results
```
